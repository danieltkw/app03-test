-- getOrderProducts
SELECT order_line.*, 
       db_products.sku, 
       db_products.price, 
       db_products.image, 
       `order`.total, 
       `order`.taxes, 
       `order`.subtotal, 
       db_products.name
FROM order_line 
INNER JOIN db_products ON order_line.product_id = db_products.product_id
INNER JOIN `order` ON order_line.order_id = `order`.order_id
WHERE order_line.order_id = {{appsmith.store.order.Order_id}};
